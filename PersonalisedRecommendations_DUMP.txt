'use client'
// components/PersonalisedRecommendations.tsx
// Shows personalised product recommendations based on browsing history.

import { useEffect, useState } from 'react'
import { getPreferredCategories, getPreferredBrands } from '@/lib/intelligence/personalisation'
import { INDIA_CATALOG } from '@/lib/india-catalog'
import ProductCard from './ProductCard'

export default function PersonalisedRecommendations() {
  const [products, setProducts] = useState<any[]>([])
  const [heading, setHeading] = useState('Popular Right Now')

  useEffect(() => {
    try {
      const cats = getPreferredCategories()
      const brands = getPreferredBrands()
      if (cats.length > 0 || brands.length > 0) {
        const matched = INDIA_CATALOG.filter(p =>
          cats.includes(p.category) || brands.includes(p.brand)
        ).slice(0, 8)
        if (matched.length >= 4) { setProducts(matched); setHeading('Picked For You') ; return }
      }
      setProducts(INDIA_CATALOG.filter(p => p.rating && p.rating >= 4.3).slice(0, 8))
    } catch {
      setProducts(INDIA_CATALOG.slice(0, 8))
    }
  }, [])

  if (products.length === 0) return null

  return (
    <section className="py-10">
      <div className="mx-auto max-w-7xl px-4 sm:px-6 lg:px-8">
        <h2 className="text-2xl font-black tracking-tighter mb-6">{heading}</h2>
        <div className="grid grid-cols-2 sm:grid-cols-3 lg:grid-cols-4 gap-4">
          {products.map(p => <ProductCard key={p.id} product={p} />)}
        </div>
      </div>
    </section>
  )
}

    import Redis from 'ioredis'
    import { isConfigured } from '@/lib/env'
    let client: Redis | null = null
    export function getRedis(): Redis | null {
      if (!isConfigured('REDIS_URL')) return null
      if (!client) client = new Redis(process.env.REDIS_URL!, { lazyConnect: true, enableReadyCheck: false })
      return client
    }
    export async function cacheGet(key: string): Promise<string | null> {
      return (await getRedis()?.get(key)) ?? null
    }
    export async function cacheSet(key: string, value: string, ttl = 3600): Promise<void> {
      await getRedis()?.set(key, value, 'EX', ttl)
    }
    export async function cacheDel(key: string): Promise<void> {
      await getRedis()?.del(key)
    }
